## 使用方式

### CommonJS

```js
const { search, song_url, lyric } = require('NeteaseCloudMusicApi')

// 搜索歌曲
const result = await search({ keywords: '周杰伦' })

// 获取歌曲播放链接
const url = await song_url({ id: result.body.result.songs[0].id })

// 获取歌词
const lyrics = await lyric({ id: result.body.result.songs[0].id })
```

### ES Module (TypeScript)

```ts
import { search, song_url, lyric } from 'NeteaseCloudMusicApi'

const result = await search({ keywords: '周杰伦', type: 1 })
const url = await song_url({ id: result.body.result.songs[0].id, br: 320000 })
const lyrics = await lyric({ id: result.body.result.songs[0].id })
```
